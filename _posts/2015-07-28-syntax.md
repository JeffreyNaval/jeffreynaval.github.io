PHP

```php
<?php

// Person Class
class Person
{
    public $name;

    public function __construct($name)
    {
        $this->name = $name;
    }

    public function greet()
    {
        return 'Hello, my name is ' . $this->name;
    }
}
```

import "fmt"

    x, ok := <- ch

    go println(len("hello world!"))
    return
}
